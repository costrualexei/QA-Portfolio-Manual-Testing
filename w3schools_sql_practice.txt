/* Verificarea clienților dintr-o anumită țară */
SELECT CustomerName, ContactName, City 
FROM Customers 
WHERE Country = 'Mexico' 
ORDER BY CustomerName ASC;

/* Numărarea produselor pe categorii */
SELECT CategoryID, COUNT(ProductID) AS TotalProducts 
FROM Products 
GROUP BY CategoryID;


/* Identificarea comenzilor și a angajaților care le-au procesat */
SELECT Orders.OrderID, Customers.CustomerName, Employees.LastName AS EmployeeName
FROM Orders
INNER JOIN Customers ON Orders.CustomerID = Customers.CustomerID
INNER JOIN Employees ON Orders.EmployeeID = Employees.EmployeeID
LIMIT 10;

/* Căutarea contactelor care încep cu litera M */
SELECT * 
FROM Customers 
WHERE ContactName LIKE 'M%';

/* Găsește toți clienții din Brazilia care locuiesc în São Paulo */
SELECT CustomerName, ContactName, City, Country 
FROM Customers 
WHERE Country = 'Brazil' AND City = 'São Paulo';


-- Numără câte comenzi a procesat fiecare angajat, sortat de la cel mai activ
SELECT EmployeeID, COUNT(OrderID) AS TotalOrders
FROM Orders
GROUP BY EmployeeID
ORDER BY TotalOrders DESC;

-- Selectează furnizorii din UK, USA și Japonia
SELECT SupplierName, ContactName, Country 
FROM Suppliers 
WHERE Country IN ('UK', 'USA', 'Japan');

-- Unește 3 tabele pentru a vedea detaliile complete ale expedierii
SELECT 
    Orders.OrderID, 
    Customers.CustomerName, 
    Shippers.ShipperName AS DeliveryCompany,
    Orders.OrderDate
FROM Orders
JOIN Customers ON Orders.CustomerID = Customers.CustomerID
JOIN Shippers ON Orders.ShipperID = Shippers.ShipperID
ORDER BY Orders.OrderDate DESC
LIMIT 15;
